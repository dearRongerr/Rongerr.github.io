def box_corner_to_center(boxes):
    """‰ªéÔºàÂ∑¶‰∏äÔºåÂè≥‰∏ãÔºâËΩ¨Êç¢Âà∞Ôºà‰∏≠Èó¥ÔºåÂÆΩÂ∫¶ÔºåÈ´òÂ∫¶Ôºâ"""

    cx = (x1 + x2) / 2
    cy = (y1 + y2) / 2
    w = x2 - x1
    h = y2 - y1
    boxes = np.stack((cx, cy, w, h), axis=-1)
    return boxes

#@save
def box_center_to_corner(boxes):
    """‰ªéÔºà‰∏≠Èó¥ÔºåÂÆΩÂ∫¶ÔºåÈ´òÂ∫¶ÔºâËΩ¨Êç¢Âà∞ÔºàÂ∑¶‰∏äÔºåÂè≥‰∏ãÔºâ"""

    y1 = cy - 0.5 * h
    x2 = cx + 0.5 * w
    y2 = cy + 0.5 * h
    boxes = np.stack((x1, y1, x2, y2), axis=-1)
    return boxes

def multibox_prior(data, sizes, ratios):
    """ÁîüÊàê‰ª•ÊØè‰∏™ÂÉèÁ¥†‰∏∫‰∏≠ÂøÉÂÖ∑Êúâ‰∏çÂêåÂΩ¢Áä∂ÁöÑÈîöÊ°Ü"""
    in_height, in_width = data.shape[-2:] # b c h w
    device, num_sizes, num_ratios = data.device, len(sizes), len(ratios)
    boxes_per_pixel = (num_sizes + num_ratios - 1)
    # boxes_per_pixel = num_sizes * num_ratios
    size_tensor = torch.tensor(sizes, device=device) # sizesÔºöscale input h & w ÂàÜÂà´Áº©Êîæ size
    ratio_tensor = torch.tensor(ratios, device=device) # ratioÔºöaspect ratio

    # ‰∏∫‰∫ÜÂ∞ÜÈîöÁÇπÁßªÂä®Âà∞ÂÉèÁ¥†ÁöÑ‰∏≠ÂøÉÔºåÈúÄË¶ÅËÆæÁΩÆÂÅèÁßªÈáè„ÄÇ
    # Âõ†‰∏∫‰∏Ä‰∏™ÂÉèÁ¥†ÁöÑÈ´ò‰∏∫1‰∏îÂÆΩ‰∏∫1ÔºåÊàë‰ª¨ÈÄâÊã©ÂÅèÁßªÊàë‰ª¨ÁöÑ‰∏≠ÂøÉ0.5
    offset_h, offset_w = 0.5, 0.5
    steps_h = 1.0 / in_height  # Âú®yËΩ¥‰∏äÁº©ÊîæÊ≠•Èïø
    steps_w = 1.0 / in_width  # Âú®xËΩ¥‰∏äÁº©ÊîæÊ≠•Èïø

    # ÁîüÊàêÈîöÊ°ÜÁöÑÊâÄÊúâ‰∏≠ÂøÉÁÇπ
    # Âõ†‰∏∫ÂéüÁÇπÂú®Â∑¶‰∏äËßíÔºåÊâÄ‰ª•ÊòØÈîöÊ°Ü‰∏≠ÂøÉÁÇπÂàÜÂà´Âä†offset_h„ÄÅoffset_w
    center_h = (torch.arange(in_height, device=device) + offset_h) * steps_h
    center_w = (torch.arange(in_width, device=device) + offset_w) * steps_w
    # ÂØπÊØè‰∏™ÂÉèÁ¥†ÁÇπÁîüÊàêÈîöÊ°ÜÔºåÊØè‰∏™ÂÉèÁ¥†ÁÇπÈÉΩÊúâÁîüÊàê num_sizes+num_ratios-1ÁöÑÈîöÊ°ÜÔºåÊâÄ‰ª•ÂØπ‰∫éÂçï‰∏™ÂÉèÁ¥†ÁÇπÊù•ËØ¥
    # ÂèÇËÄÉÁÇπÂèò‰∫ÜÔºåÂÖ®ÈÉ®ÂèòÊàê‰∫ÜÂçï‰ΩçÂùêÊ†á‰∏ãÁöÑÂ∫¶ÈáèÔºåÊâÄ‰ª•Èô§‰ª• steps_h Âíå steps_w
    shift_y, shift_x = torch.meshgrid(center_h, center_w, indexing='ij')
    shift_y, shift_x = shift_y.reshape(-1), shift_x.reshape(-1)

    # ÁîüÊàê‚Äúboxes_per_pixel‚Äù‰∏™È´òÂíåÂÆΩÔºå
    # ‰πãÂêéÁî®‰∫éÂàõÂª∫ÈîöÊ°ÜÁöÑÂõõËßíÂùêÊ†á(xmin,xmax,ymin,ymax)
    w = torch.cat((size_tensor * torch.sqrt(ratio_tensor[0]),
                   sizes[0] * torch.sqrt(ratio_tensor[1:])))\
                   * in_height / in_width  # Â§ÑÁêÜÁü©ÂΩ¢ËæìÂÖ•
    h = torch.cat((size_tensor / torch.sqrt(ratio_tensor[0]),
                   sizes[0] / torch.sqrt(ratio_tensor[1:])))
    # Èô§‰ª•2Êù•Ëé∑ÂæóÂçäÈ´òÂíåÂçäÂÆΩ
    anchor_manipulations = torch.stack((-w, -h, w, h)).T.repeat(
                                        in_height * in_width, 1) / 2

    # ÊØè‰∏™‰∏≠ÂøÉÁÇπÈÉΩÂ∞ÜÊúâ‚Äúboxes_per_pixel‚Äù‰∏™ÈîöÊ°ÜÔºå
    # ÊâÄ‰ª•ÁîüÊàêÂê´ÊâÄÊúâÈîöÊ°Ü‰∏≠ÂøÉÁöÑÁΩëÊ†ºÔºåÈáçÂ§ç‰∫Ü‚Äúboxes_per_pixel‚ÄùÊ¨°
    out_grid = torch.stack([shift_x, shift_y, shift_x, shift_y],
                dim=1).repeat_interleave(boxes_per_pixel, dim=0)
    output = out_grid + anchor_manipulations
    return output.unsqueeze(0)

def box_iou(boxes1, boxes2):
    """ËÆ°ÁÆó‰∏§‰∏™ÈîöÊ°ÜÊàñËæπÁïåÊ°ÜÂàóË°®‰∏≠ÊàêÂØπÁöÑ‰∫§Âπ∂ÊØî"""
    box_area = lambda boxes: ((boxes[:, 2] - boxes[:, 0]) *
                              (boxes[:, 3] - boxes[:, 1]))
    # boxes1,boxes2,areas1,areas2ÁöÑÂΩ¢Áä∂:
    # boxes1Ôºö(boxes1ÁöÑÊï∞Èáè,4),
    # boxes2Ôºö(boxes2ÁöÑÊï∞Èáè,4),
    # areas1Ôºö(boxes1ÁöÑÊï∞Èáè,),
    # areas2Ôºö(boxes2ÁöÑÊï∞Èáè,)
    areas1 = box_area(boxes1)
    areas2 = box_area(boxes2)
    # inter_upperlefts,inter_lowerrights,intersÁöÑÂΩ¢Áä∂:
    # (boxes1ÁöÑÊï∞Èáè,boxes2ÁöÑÊï∞Èáè,2)
    inter_upperlefts = torch.max(boxes1[:, None, :2], boxes2[:, :2])
    inter_lowerrights = torch.min(boxes1[:, None, 2:], boxes2[:, 2:])
    inters = (inter_lowerrights - inter_upperlefts).clamp(min=0)
    # inter_areasandunion_areasÁöÑÂΩ¢Áä∂:(boxes1ÁöÑÊï∞Èáè,boxes2ÁöÑÊï∞Èáè)
    inter_areas = inters[:, :, 0] * inters[:, :, 1]
    union_areas = areas1[:, None] + areas2 - inter_areas
    return inter_areas / union_areas
