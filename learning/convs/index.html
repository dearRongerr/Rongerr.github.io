def matrix_multiplication_for_conv2d(input,kernel,bias=0,stride=1,padding=0):

  if padding >0:
    input = F.pad(input,(padding,padding,padding,padding))


  input_h,input_w = input.shape
  kernel_h,kernel_w = kernel.shape

  output_h = (math.floor((input_h - kernel_h)/stride) + 1)  # Âç∑ÁßØËæìÂá∫ÁöÑÈ´òÂ∫¶
  output_w = (math.floor((input_w - kernel_w)/stride) + 1)  # Âç∑ÁßØËæìÂá∫ÁöÑÂÆΩÂ∫¶ 
  output = torch.zeros(output_h,output_w) # ÂàùÂßãÂåñ ËæìÂá∫Áü©Èòµ

  for i in range(0,input_h - kernel_h + 1,stride): # ÂØπÈ´òÂ∫¶ËøõË°åÈÅçÂéÜ

  output_w = (math.floor((input_w - kernel_w)/stride) + 1)  # Âç∑ÁßØËæìÂá∫ÁöÑÂÆΩÂ∫¶ 
  output = torch.zeros(output_h,output_w) # ÂàùÂßãÂåñ ËæìÂá∫Áü©Èòµ

  kernel_matrix = kernel.reshape(kernel.numel(),1) # Â≠òÂÇ®ÁùÄkernelÁöÑ ÂàóÂêëÈáèÔºàÁü©ÈòµÔºâÂΩ¢Âºè
  row_index = 0

  for i in range(0,input_h - kernel_h + 1,stride): # ÂØπÈ´òÂ∫¶ËøõË°åÈÅçÂéÜ
    for j in range(0,input_w - kernel_w +1,stride):  # ÂØπÂÆΩÂ∫¶Áª¥ËøõË°åÈÅçÂéÜ

      region_vector = torch.flatten(region)
      region_matrix[row_index] = region_vector
      row_index +=1

  output_matrix = region_matrix @ kernel_matrix
  output = output_matrix.reshape((output_h,output_w))+bias

  return output

  output_w = (math.floor((input_w - kernel_w)/stride) + 1)  # Âç∑ÁßØËæìÂá∫ÁöÑÂÆΩÂ∫¶

    def matrix_multiplication_for_conv2d_finall(input,kernel,bias=None,stride=1,padding=0,dilation=1,groups=1):
    if padding>0:
        input = F.pad(input,(padding,padding,padding,padding,0,0,0,0))

    bs,in_channel,input_h,input_w = input.shape
    out_channel,_,kernel_h,kernel_w = kernel.shape

    assert out_channel % groups == 0 and in_channel % groups==0,"groupsÂøÖÈ°ªË¶ÅÂêåÊó∂Ë¢´ËæìÂÖ•ÈÄöÈÅìÂíåËæìÂá∫ÈÄöÈÅìÊï∞Êï¥Èô§ÔºÅ"
    input = input.reshape((bs,groups,in_channel//groups,input_h,input_w))
    kernel = kernel.reshape((groups,out_channel//groups,in_channel//groups,kernel_h,kernel_w))

    kernel_h = (kernel_h-1)*(dilation-1)+kernel_h
    kernel_w = (kernel_w-1)*(dilation-1)+kernel_w

    output_h = math.floor((input_h-kernel_h)/stride)+1
    output_w = math.floor((input_w-kernel_w)/stride)+1

    output_shape = (bs,groups,out_channel//groups,output_h,output_w)
    output = torch.zeros(output_shape)

    if bias is None:
        bias = torch.zeros(out_channel)

    for ind in range(bs): # ÂØπbatch sizeËøõË°åÈÅçÂéÜ
        for g in range(groups): # ÂØπÁæ§ÁªÑËøõË°åÈÅçÂéÜ
            for oc in range(out_channel//groups): # ÂØπÂàÜÁªÑÂêéÁöÑËæìÂá∫ÈÄöÈÅìËøõË°åÈÅçÂéÜ
                for ic in range(in_channel//groups): # ÂØπÂàÜÁªÑÂêéÁöÑËæìÂÖ•ÈÄöÈÅìËøõË°åÈÅçÂéÜ
                    for i in range(0,input_h-kernel_h+1,stride): #ÂØπÈ´òÂ∫¶ÈÅçÂéÜ
                        for j in range(0,input_w-kernel_w+1,stride): # ÂØπÂÆΩÂ∫¶ÈÅçÂéÜ
                            region = input[ind,g,ic,i:i+kernel_h:dilation,j:j+kernel_w:dilation] #ÁâπÂæÅÂå∫Âüü

                output[ind,g,oc] += bias[g*(out_channel//groups)+oc]  # ËÄÉËôëÂÅèÁΩÆÈ°π
    output = output.reshape((bs,out_channel,output_h,output_w))  # ËøòÂéüÊàêÂõõÁª¥Âº†Èáè
    return output
